cardapio
- idCardapio (int) PK
- nome (string)
- preco	(númerico decimal)
- tipo (string)
- descricao (string)
- foto	(string)
- status (boolean)

CREATE TABLE cardapio(
	idCardapio int PRIMARY KEY auto_increment,
	nome varchar(100),
	preco decimal(10,2),
	tipo varchar(100),
	descricao text,
	foto varchar(100),
	status boolean
) engine=InnoDB;

O sistema de avaliação deverá estar integrado ao cardápio, permitindo que possa ser feito:

A avaliação do prato com escala de 1 a 5 estrelas
Comentário adicional. Neste caso, alem do campo de comentário, será necessário incluir a data, o nome, e o e-mail da pessoa


CREATE TABLE avaliacaoes(
	idAvaliacao int PRIMARY KEY auto_increment,
	nota int,
	comentario text,
	data date,
	nome varchar(100),
	email varchar(100),
	situacao varchar(50),
	idCardapio int,
	FOREIGN KEY (idCardapio) REFERENCES cardapio(idCardapio)
) engine=InnoDB;
